  This code creates a typewriter effect that displays a set of lines character 
  by character. It starts when a button is clicked and can be used to 
  gradually reveal text to the user.

  //Array of text lines to be displayed
  const lines = [
  // the text
  ];

  // Speed at which characters are displayed (in milliseconds)
  const speed = 50;

  // Initialize character index
  let charIndex = 0;

  // Function to create the typewriter effect
  function typeWriter() {
    // Get elements where text should be displayed
    const lineElements = lines.map((_, index) => document.getElementById(`line${index + 1}`));
    // Function to write a single character
    function writeCharacter() {
      lineElements.forEach((element, index) => {
        if (charIndex < lines[index].length) {
          element.textContent += lines[index].charAt(charIndex);
        }
      });
      charIndex++;
      if (charIndex < lines[0].length) {
        setTimeout(writeCharacter, speed);
      }
    }
    // Start the typewriter effect
    setTimeout(writeCharacter, speed);
  }
  // Function to start the typewriter effect when a button is clicked
  function startTypewriter(button) {
    // Hide the button
    button.style.display = "none";
    
    // Start the typewriter effect
    typeWriter();
    // Display additional content (if needed)
    document.getElementById("reference").style.display = "block";
  }
